% Stretching table that fills a given height.
% #1=title, #2=ncols, #3=colspec, #4=height, #5=rows
\newcommand{\InvTable}[5]{%
  \noindent\vbox to #4{%
    \begingroup%
    \setlength{\tabcolsep}{5pt}%
    \arrayrulecolor{InvLine}%
    \setlength{\arrayrulewidth}{0.5pt}%
    \def\R##1{##1\\\hline}%
    \begin{tabularx}{\linewidth}{|#3|}
      \hline%
      \rowcolor{InvBlueSoft}\multicolumn{#2}{|l|}{\InvHeader #1} \\%
      \hline%
      #5%
    \end{tabularx}%
    \endgroup%
    \vfil%
  }%
}
