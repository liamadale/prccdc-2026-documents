\colorsec{secteal}{Linux --- System \& OS}
\begin{lstlisting}
hostname; hostname -f
cat /etc/os-release; uname -a
hostnamectl 2>/dev/null
uptime; who -a; last -n 10
timedatectl 2>/dev/null; df -hT
\end{lstlisting}

\colorsec{secteal}{Linux --- Network \& Exposure}
\begin{lstlisting}
ip a; ip r; ip neigh
resolvectl status 2>/dev/null || cat /etc/resolv.conf
ss -tulpen                # listeners (ports/PIDs)
lsof -nP -iTCP -sTCP:LISTEN
sudo iptables -S 2>/dev/null
sudo ufw status verbose 2>/dev/null
sudo firewall-cmd --list-all 2>/dev/null
\end{lstlisting}

\colorsec{secpurple}{Linux --- Users, Privilege, Auth}
\begin{lstlisting}
id; groups; sudo -l
getent passwd | cut -d: -f1,3,6,7 | column -t -s:
getent group | head
awk -F: '$3==0{print}' /etc/passwd
grep -R "PermitRootLogin\|PasswordAuthentication" /etc/ssh/sshd_config*
\end{lstlisting}

\colorsec{secgreen}{Linux --- Services, Tasks, Logs}
\begin{lstlisting}
systemctl --type=service --state=running
systemctl list-timers --all
crontab -l; sudo crontab -l
ls -la /etc/cron.* /var/spool/cron 2>/dev/null
journalctl -p warning..alert -n 100 --no-pager
\end{lstlisting}

\colorsec{secyellow}{Linux --- Packages \& Patch Level}
\begin{lstlisting}
dpkg -l | head; apt list --upgradable 2>/dev/null | head
rpm -qa | head; dnf check-update 2>/dev/null | head
\end{lstlisting}

\colorsec{secorange}{Linux --- Data, Databases, Shares}
\begin{lstlisting}
mysql -u root -e "SHOW DATABASES;" 2>/dev/null
grep -r "datadir" /etc/mysql/ /etc/my.cnf 2>/dev/null
sudo -u postgres psql -c "\l" 2>/dev/null
cat /etc/exports 2>/dev/null       # NFS
testparm -s 2>/dev/null | grep path # Samba
find / -name "*.pem" -o -name "*.key" 2>/dev/null
find /var/www /srv -name "*.conf" -exec grep -il "pass" {} \; 2>/dev/null
\end{lstlisting}

\colorsec{secred}{Linux --- Security Controls \& Telemetry}
\begin{lstlisting}
getenforce 2>/dev/null; aa-status 2>/dev/null
auditctl -l 2>/dev/null
systemctl is-active auditd rsyslog syslog-ng 2>/dev/null
ls /var/log/; journalctl --disk-usage
\end{lstlisting}

\columnbreak

\colorsec{secteal}{Windows --- System \& OS (PowerShell)}
\begin{lstlisting}
$env:COMPUTERNAME
systeminfo | findstr /B /C:"OS Name" /C:"OS Version"
Get-ComputerInfo | select OsName,OsVersion,WindowsBuildLabEx
Get-TimeZone; w32tm /query /status
Get-Volume | ft -AutoSize
\end{lstlisting}

\colorsec{secteal}{Windows --- Network \& Exposure}
\begin{lstlisting}
Get-NetIPAddress | sort InterfaceAlias | ft -AutoSize
Get-NetRoute -DestinationPrefix "0.0.0.0/0" | ft -AutoSize
Get-DnsClientServerAddress | ft -AutoSize
Get-NetTCPConnection -State Listen | sort LocalPort | ft -AutoSize
Get-NetFirewallProfile | ft -AutoSize
Get-NetFirewallRule -Enabled True | select -First 20
\end{lstlisting}

\colorsec{secpurple}{Windows --- Users, Groups, Remote Access}
\begin{lstlisting}
whoami /all
Get-LocalUser | ft -AutoSize
Get-LocalGroupMember Administrators
net user; net localgroup administrators
Get-ItemProperty "HKLM:\SYSTEM\CurrentControlSet\Control\Terminal Server" -Name fDenyTSConnections
Get-NetTCPConnection -State Listen | ? {$_.LocalPort -in 3389,5985,5986}
\end{lstlisting}

\colorsec{secgreen}{Windows --- Services, Tasks, Logs}
\begin{lstlisting}
Get-Service | ? {$_.Status -eq "Running"} | sort Name
Get-ScheduledTask | select TaskName,State | ft -AutoSize
schtasks /query /fo LIST /v | more
Get-WinEvent -LogName Security -MaxEvents 30
\end{lstlisting}

\colorsec{secyellow}{Windows --- Patch Level \& Roles}
\begin{lstlisting}
Get-HotFix | sort InstalledOn -Descending | select -First 10
Get-WindowsFeature | ? InstallState -eq Installed
nltest /dsgetdc: 2>$null           # AD/DC check
\end{lstlisting}

\colorsec{secorange}{Windows --- Data, Databases, Shares}
\begin{lstlisting}
sqlcmd -S localhost -Q "SELECT name,physical_name FROM sys.master_files" 2>$null
Get-Service *SQL* | ft -AutoSize
Get-SmbShare | ft -AutoSize; net share
Get-IISSite 2>$null | select Name,PhysicalPath
Get-ChildItem C:\ -Recurse -Include *.pfx,*.p12,*.key -EA SilentlyContinue
dir C:\inetpub /s /b 2>$null | findstr /i "web.config"
\end{lstlisting}

\colorsec{secred}{Windows --- Security Controls \& Telemetry}
\begin{lstlisting}
Get-MpComputerStatus | select *Enabled,*Updated
Get-Service WinDefend,Sysmon*,WinRM | ft -AutoSize
auditpol /get /category:*
Get-ItemProperty "HKLM:\SOFTWARE\Policies\Microsoft\Windows\PowerShell\ScriptBlockLogging" 2>$null
wevtutil el | findstr /i "security sysmon powershell"
\end{lstlisting}
